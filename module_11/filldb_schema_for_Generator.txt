


CREATE TABLE departments (department_id INT NOT NULL AUTO_INCREMENT, department_name VARCHAR(30) NOT NULL,PRIMARY KEY(department_id));

CREATE TABLE employee (personnel_id INT NOT NULL AUTO_INCREMENT,f_name    VARCHAR(20) NOT NULL,l_name   VARCHAR(20) NOT NULL,department_id INT NOT NULL, PRIMARY KEY(personnel_id), CONSTRAINT fk_department FOREIGN KEY(department_id) REFERENCES departments(department_id));

CREATE TABLE suppliers (supplier_id INT NOT NULL AUTO_INCREMENT, supplier_name VARCHAR(40) NOT NULL, supplier_address VARCHAR(60) NOT NULL, supplier_contact VARCHAR(20) NOT NULL, PRIMARY KEY(supplier_id));

CREATE TABLE supplies (item_id INT NOT NULL AUTO_INCREMENT, supply_name VARCHAR(20) NOT NULL, supplier_id INT NOT NULL, PRIMARY KEY(item_id), CONSTRAINT fk_suppliers FOREIGN KEY(supplier_id) REFERENCES suppliers(supplier_id));

CREATE TABLE employee_time (record_id INT NOT NULL AUTO_INCREMENT, personnel_id INT NOT NULL, work_date date, hours INT, PRIMARY KEY(record_id));

CREATE TABLE deliveries_log (shipment_id INT NOT NULL AUTO_INCREMENT, expected_delivery_date date, actual_delivery_date date, supplier_id INT NOT NULL, PRIMARY KEY(shipment_id));

CREATE TABLE wine_distribution (order_id INT NOT NULL AUTO_INCREMENT, order_date date, wine_name VARCHAR(20), order_units INT NOT NULL, distributor_name VARCHAR(100), PRIMARY KEY(order_id));

CREATE TABLE wine_production (batch_id INT NOT NULL AUTO_INCREMENT, wine_name VARCHAR(20), born_date date, units_produced INT, PRIMARY KEY(batch_id));
