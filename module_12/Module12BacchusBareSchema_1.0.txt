CREATE TABLE departments (department_id INT NOT NULL AUTO_INCREMENT, department_name VARCHAR(30) NOT NULL,PRIMARY KEY(department_id));

CREATE TABLE employee (employee_id INT NOT NULL AUTO_INCREMENT, f_name    VARCHAR(20) NOT NULL, l_name VARCHAR(20) NOT NULL, department_id INT NOT NULL, PRIMARY KEY(employee_id), CONSTRAINT fk_department FOREIGN KEY(department_id) REFERENCES departments(department_id));

CREATE TABLE suppliers (supplier_id INT NOT NULL AUTO_INCREMENT, supplier_name VARCHAR(40) NOT NULL, supplier_address VARCHAR(60) NOT NULL, supplier_contact VARCHAR(20) NOT NULL, PRIMARY KEY(supplier_id));

CREATE TABLE supplies (item_id INT NOT NULL AUTO_INCREMENT, supply_name VARCHAR(20) NOT NULL, supplier_id INT NOT NULL, PRIMARY KEY(item_id), CONSTRAINT fk_suppliers FOREIGN KEY(supplier_id) REFERENCES suppliers(supplier_id));

CREATE TABLE deliveries_log (shipment_id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,expected_delivery_date date,actual_delivery_date date,supplier_id INT NOT NULL);

CREATE TABLE wine_distribution (order_id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,order_date date,wine_name VARCHAR(20), order_units INT NOT NULL,distributor_name VARCHAR(100));

CREATE TABLE wine_production (batch_id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,wine_name VARCHAR(20),born_date date,units_produced INT);

CREATE TABLE employee_clocks (clock_id INT NOT NULL AUTO_INCREMENT,employee_id INT NOT NULL, department_id INT,clock_in DATETIME,clock_out DATETIME,PRIMARY KEY(clock_id),CONSTRAINT fk_employee FOREIGN KEY(employee_id) REFERENCES employee(employee_id), CONSTRAINT fk_departments FOREIGN KEY(department_id) REFERENCES departments(department_id));
